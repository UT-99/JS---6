            var Minimal_Angle = (h, m) => 
            {
                   let hourAngel = ((h * 360)/12) + ((m * 360) / (12 * 60));
                   let minAngel = (m * 360) / 60;
                   angel = Math.abs(hourAngel - minAngel);
                   if (angel > 180){

            };

Ques. 3 Perfect Number Check. 

 
        var Perfect_Check = (N) => 
            {
                  var sum = 0 ;  
                  if (N === 1)
                    return 'YES';
                  else {
                    for ( let  i = 1 ; i <= parseInt( N / 2 ) ; i++){
                      if ( N % i === 0){
                        sum = sum + i;
                      }
                    }
                    if (sum === N ){
                      return 'YES'
                    }
                    else 
                     return 'NO'
                  }
                    
            };


Ques. 4 Reverse a Number. 

 
        var Reverse_Number = (N) => 
            {
              var temp = N;
              var quotient = N;
              var remainder = 0 ;
              var reversedNumber = 0 ;
              
              while (quotient > 0 ){
                remainder = quotient % 10 ;
                reversedNumber = ( reversedNumber * 10 ) + remainder;
                quotient = parseInt(quotient / 10);
              }
            
              return reversedNumber;
            
            };
